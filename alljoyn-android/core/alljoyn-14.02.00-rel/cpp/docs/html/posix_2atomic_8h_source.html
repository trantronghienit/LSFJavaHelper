00047 inline int32_t IncrementAndFetch(volatile int32_t* mem)
00048 {
00049     /*
00050      * Androids built in __atomic_inc operation returns the previous value
00051      * stored in mem. This value has been changed in memory.  Since we expect
00052      * the value after the increment to be returned we add one to the value
00053      * returned by __atomic_inc.
00054      */
00055     return __atomic_inc(mem) + 1;
00056 }

00064 inline int32_t DecrementAndFetch(volatile int32_t* mem)
00065 {

00067      * Androids built in __atomic_dec operation returns the previous value

00069      * the value after the decrement to be returned we subtract one from the
00070      * value returned by __atomic_dec.
00071      */
